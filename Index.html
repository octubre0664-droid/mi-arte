        <a-entity mindar-image-target="targetIndex: 0" id="cuadro-target">

    <script>

        const boton = document.getElementById('boton-comenzar');

        const pantalla = document.getElementById('pantalla-inicio');

        const escena = document.querySelector('a-scene');

        const targetEl = document.querySelector('#cuadro-target');

        const entitySound = document.querySelector('#controlador-sonido');

        const audioElemento = document.getElementById('sonidoCuadro');


        // Al hacer clic en el botón, forzamos al navegador a despertar la cámara y el motor de audio

        boton.addEventListener('click', () => {

            pantalla.style.display = 'none'; // Escondemos el menú

            

            // Inicializamos el motor de Realidad Aumentada

            escena.systems["mindar-image-system"].start(); 

            

            // Forzamos al contexto de audio a "despertar" (Regla estricta de iOS/Android)

            audioElemento.play().then(() => {

                audioElemento.pause(); // Lo reproducimos y pausamos de inmediato para dejarlo listo

            }).catch(e => console.log("Audio precargado listo"));

        });


        // Evento: La cámara encuentra tu cuadro

        targetEl.addEventListener("targetFound", () => {

            console.log("Cuadro detectado");

            entitySound.components.sound.playSound();

        });


        // Evento: La cámara pierde de vista el cuadro

        targetEl.addEventListener("targetLost", () => {

            console.log("Cuadro perdido");

            entitySound.components.sound.pauseSound();

        });

    </script>
